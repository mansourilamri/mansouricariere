log_batch_size = Integer(
  Entrées: Une série de notes.
#
# Sorties: Affiche une note littérale selon la moyenne obtenue.
#
# Description: Le script demande à l'utilisateur d'entrée des notes dans 2
# tableaux. Au total, 10 notes seront entrées. On calcule la
# moyenne de ces notes et on l'affiche.
#
# Modifications:
#
###############################################################################
#
### Variables
#
[int] $somme = 0 # La somme des notes reçues.
[float] $moyenne = 0 # La moyenne des notes.
[int[]] $notes1 = @() # Premier tableau des notes.
[int[]] $notes2 = @() # Second tableau des notes.
[int[]] $notes3 = @() # Tableau combiné des notes.
[char] $noteLitterale # Note littérale selon la moyenne.
[string] $mention # Mention obtenue.
$i = 0 # variable de boucle.
##############################################################################
# MAIN #
##############################################################################
Clear-Host
#
### Acquisition des données du tableau 1.
#
do {
Write-Host "Entrez une note pour le premier tableau."
$notes1 += Read-Host
$i++
} while ( $i -lt 5 )
#
### Ajout de 2 données au tableau 1.
#
$i = 0 # Reset de la valeur.
do {
Write-Host "Ajoutez une note au tableau 1."
$notes1 += Read-Host
$i++
} while ( $i -lt 2 )
#
### Acquisition des données du tableau 2.
#
$i = 0 # Reset de la valeur.
do {
Write-Host "Entrez une note pour le second tableau."
$notes2 += Read-Host
$i++
} while ( $i -lt 3 )
#
### Création du tableau 3 par concaténation des tableaux 1 et 2.
#
$notes3 = $notes1 + $notes2
#
### Calcul de la somme des notes du tableau 3.
#
$i = 0 # Reset de la valeur.
while ( $i -lt $notes3.Length ) {
$somme += $notes3[$i]
$i++
} #
### Calcul de la moyenne.
#
$moyenne = $somme / $notes3.Length
#Write-Host( "La moyenne est: $moyenne" )
#
### Determine la note littérale et la mention accordée selon la moyenne.
#
if ( $moyenne -ge 90 ) {
$noteLitterale = 'A'
$mention = "Excellent"
}
elseif (( $moyenne -ge 80 ) -and ( $moyenne -le 89 )) {
$noteLitterale = 'B'
$mention = "Tres bien"
}
elseif (( $moyenne -ge 70 ) -and ( $moyenne -le 79 )) {
$noteLitterale = 'C'
$mention = "Bien"
}
elseif (( $moyenne -ge 60 ) -and ( $moyenne -le 69 )) {
$noteLitterale = 'D'
$mention = "Passable"
}
else {
$noteLitterale = 'E'
$mention = "Echec"
} #
### Affiche la note littérale et la mention.
#
Write-Host( 'Note litterale: ' + $noteLitterale + ' mention: ' + $mention )
